\makeproblem{Hermite recurrence relations.}{problem:hermiteDerivativeRecurranceRelation:1}{
Recall that Hermite polynomials are basis functions for the SHO wave functions (\citep{desai2009quantum} eq. 9.58).  Show that
\begin{equation*}
\frac{d}{dx} H_n(x)
=
2 n H_{n-1}(x).
\end{equation*}
} % problem

\makeanswer{problem:hermiteDerivativeRecurranceRelation:1}{
We could just look up this formula, and if we did so can find it in various places, such as
\citep{abramowitz1964handbook}, or \cite[Eq.~18.9.25]{NIST:DLMF2020}.

To derive it ourself, we can start with the Rodgigues' formula definition of the Hermite polynomials
%
\begin{equation}\label{eqn:hermiteDerivativeRecurranceRelation:10}
H_n = (-1)^n e^{x^2} \frac{d^n}{dx^n} e^{-x^2}.
\end{equation}
%
Let us write \(D = d/dx\), and take the derivative of \(H_n\)
%
\begin{equation}\label{eqn:hermiteDerivativeRecurranceRelation:50}
\begin{aligned}
(-1)^n D H_n
&= D \left( e^{x^2} D^n e^{-x^2} \right) \\
&=
2 x e^{x^2} D^n e^{-x^2}
+ e^{x^2} D^n \left( - 2 x e^{-x^2} \right) \\
&=
2 x e^{x^2} D^n e^{-x^2}
+ e^{x^2} \sum_{k=0}^n \binom{n}{k} D^k (-2 x) D^{n-k} e^{-x^2} \\
&=
2 x e^{x^2} D^n e^{-x^2}
+ e^{x^2} \sum_{k=0}^1 \binom{n}{k} D^k (-2 x) D^{n-k} e^{-x^2} \\
&=
\cancel{2 x e^{x^2} D^n e^{-x^2} }
+ e^{x^2} \left(
-\cancel{2 x D^{n} e^{-x^2} }
- 2 n D^{n-1} e^{-x^2}
\right)  \\
&=
- 2 n e^{x^2} D^{n-1} e^{-x^2}.
\end{aligned}
\end{equation}
Rearranging proves the result.

} % answer
